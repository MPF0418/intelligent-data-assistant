## 6. 数据库设计

### 6.1 数据库表结构（基于PRD数据说明表）

#### 6.1.1 UC-01 上传数据文件相关表

```sql
-- 上传文件记录表
CREATE TABLE upload_file_record (
    file_id VARCHAR(36) PRIMARY KEY,           -- UUID格式
    file_name VARCHAR(255) NOT NULL,            -- 最大255字符
    file_size BIGINT NOT NULL,                  -- 单位：字节
    file_type VARCHAR(10) NOT NULL,             -- CSV、EXCEL
    upload_time DATETIME NOT NULL,               -- 自动记录当前时间
    row_count INT NOT NULL,                      -- 不包含表头
    col_count INT NOT NULL,                      -- 表头列数
    status VARCHAR(20) DEFAULT 'PENDING'         -- PENDING、READY、EXPIRED
);

-- 数据列信息表
CREATE TABLE data_column_info (
    column_id VARCHAR(36) PRIMARY KEY,           -- UUID格式
    file_id VARCHAR(36) NOT NULL,                -- 外键
    col_name VARCHAR(100) NOT NULL,              -- 来源表头
    col_type VARCHAR(20) NOT NULL,               -- STRING、NUMBER、DATE、BOOLEAN
    col_index INT NOT NULL,                      -- 从0开始
    null_count INT DEFAULT 0,                    -- 统计空值个数
    sample_values JSON,                          -- 最多5个示例
    FOREIGN KEY (file_id) REFERENCES upload_file_record(file_id)
);
```

#### 6.1.2 UC-03 会话管理相关表

```sql
-- 会话信息表
CREATE TABLE analysis_session (
    session_id VARCHAR(36) PRIMARY KEY,           -- UUID格式
    user_id VARCHAR(36),                          -- 预留字段
    data_file_id VARCHAR(36) NOT NULL,            -- 外键
    create_time DATETIME NOT NULL,
    last_active_time DATETIME NOT NULL,
    status VARCHAR(20) DEFAULT 'ACTIVE',          -- ACTIVE、CLOSED、EXPIRED
    context_turns INT DEFAULT 0,
    FOREIGN KEY (data_file_id) REFERENCES upload_file_record(file_id)
);

-- 会话上下文表
CREATE TABLE session_context (
    context_id VARCHAR(36) PRIMARY KEY,
    session_id VARCHAR(36) NOT NULL,
    data_schema JSON NOT NULL,                     -- { columns: [] }
    current_filters JSON,                          -- 当前筛选条件
    message_history JSON,                          -- 最近20轮对话
    FOREIGN KEY (session_id) REFERENCES analysis_session(session_id)
);
```

#### 6.1.3 UC-04 对话消息相关表

```sql
-- 对话消息表
CREATE TABLE chat_message (
    message_id VARCHAR(36) PRIMARY KEY,
    session_id VARCHAR(36) NOT NULL,
    role VARCHAR(20) NOT NULL,                    -- USER、ASSISTANT、SYSTEM
    content TEXT NOT NULL,
    timestamp DATETIME NOT NULL,
    status VARCHAR(20) DEFAULT 'PENDING',        -- PENDING、PROCESSING、COMPLETED、ERROR
    FOREIGN KEY (session_id) REFERENCES analysis_session(session_id)
);

-- 消息元数据表
CREATE TABLE message_metadata (
    metadata_id VARCHAR(36) PRIMARY KEY,
    message_id VARCHAR(36) NOT NULL,
```
