## 一、优化目标

解决当前系统在数据分析与图表绘制场景下输出结果不令人满意的问题，提升用户体验和系统智能化水平。

### 核心问题清单

1. **致命问题**（3 个）：
   - 意图识别与配置生成耦合度过高
   - 缺少图表类型智能推荐机制
   - 缺少数据预处理规范

2. **严重问题**（3 个）：
   - 缺少多图表联动机制
   - 缺少异常处理和人机交互设计
   - 缺少复杂查询的组合能力

3. **一般问题**（3 个）：
   - 语义映射表覆盖度不足
   - 缺少性能监控和日志分析机制
   - 缺少数据质量评估机制

### 新增需求

- **用户反馈系统**：收集用户对识别结果和图表质量的评价
- **错误收集系统**：自动记录所有报错数据
- **自动修复机制**：定时或手动触发批量修复已发现的问题

```javascript
class QueryConfigGenerator {
  // 增强容错性
  generateConfig(intentType, userInput, columns) {
    try {
      const config = this.tryGenerateConfig(intentType, userInput, columns);
      
      if (!config) {
        // 降级：尝试使用大模型 API
        return this.fallbackToLLM(intentType, userInput, columns);

      return config;

      this.errorCollector.record(error);
      return this.getDefaultConfig(intentType);
    }
  }
  
  // 推断数据预处理需求
  inferDataTransform(userInput, columns) {
    const transform = {};
    
    // 检测单位转换需求
    if (userInput.includes('万元') || userInput.includes('元')) {
      transform.unitConversion = {
        from: userInput.includes('万元') ? '万元' : '元',
        to: userInput.includes('万元') ? '元' : '万元',
        formula: userInput.includes('万元') ? 'value * 10000' : 'value / 10000',
        decimalPlaces: 2
      };
    }
    
    return Object.keys(transform).length > 0 ? transform : null;
  }
}
```

```javascript
// chartRecommender.js
class ChartRecommender {
  constructor() {
    // 数据特征到图表类型的映射
    this.featureToChart = {
      'timeSeries': { chart: 'line', reason: '展示时间序列趋势', weight: 0.9 },
      'categorical': { chart: 'bar', reason: '直观比较分类数据', weight: 0.85 },
      'proportional': { chart: 'pie', reason: '展示部分与整体关系', weight: 0.8 },
      'correlation': { chart: 'scatter', reason: '分析变量间相关性', weight: 0.75 }
    };
    
    // 意图类型到图表类型的映射
    this.intentToChart = {
      'CHART_BAR': ['bar'],
      'CHART_LINE': ['line'],
      'CHART_PIE': ['pie'],
      'CHART_SCATTER': ['scatter'],
      'QUERY_AGGREGATE': [
        { chart: 'bar', reason: '适合分组对比', weight: 0.8 },
        { chart: 'pie', reason: '适合占比分析', weight: 0.7 }
      ],
      'QUERY_FIND': [
        { chart: 'bar', reason: '突出极值对比', weight: 0.75 }
      ]
    };
  }
  
  // 推荐图表
  recommend(dataInfo, intentResult) {
    const recommendations = [];
    
    // 1. 分析数据特征
    const dataFeatures = this.analyzeDataFeatures(dataInfo);
    
    // 2. 基于特征推荐
    for (const feature of dataFeatures) {
      const rec = this.featureToChart[feature];
      if (rec) {
        recommendations.push({
          chartType: rec.chart,
          reason: rec.reason,
          confidence: rec.weight,
          source: 'data_feature'
        });
      }
    }
    
    // 3. 基于意图推荐
    const intentRecs = this.intentToChart[intentResult.intent] || [];
    for (const rec of intentRecs) {
      const chartType = typeof rec === 'string' ? rec : rec.chart;
      const reason = typeof rec === 'string' ? '常用图表类型' : rec.reason;
      const weight = typeof rec === 'string' ? 0.6 : rec.weight;
      
      recommendations.push({
        chartType,
        reason,
        confidence: weight,
        source: 'intent_type'
      });
    }
    
    // 4. 去重并排序
    return this.deduplicateAndSort(recommendations);
  }
  
  // 分析数据特征
  analyzeDataFeatures(dataInfo) {
    const features = [];
    const { columns, sampleData } = dataInfo;
    
    // 检测时间序列
    if (columns.some(c => /日期 | 时间 | 年份 | 月份/.test(c))) {
      features.push('timeSeries');
    }
    
    // 检测分类数据
    if (columns.some(c => /省 | 市 | 部门 | 类别 | 类型/.test(c))) {
      features.push('categorical');
    }
    
    // 检测占比数据
    if (columns.length >= 2) {
      // 有分组列和数值列
      const hasGroup = columns.some(c => /省 | 市 | 部门 | 类别/.test(c));
      const hasValue = columns.some(c => /金额 | 数量 | 数值/.test(c));
      if (hasGroup && hasValue) {
        features.push('proportional');
      }
    }
    
    return features;
  }
  
  // 去重并排序
  deduplicateAndSort(recommendations) {
    const deduped = new Map();
    
    for (const rec of recommendations) {
      if (!deduped.has(rec.chartType) || deduped.get(rec.chartType).confidence < rec.confidence) {
        deduped.set(rec.chartType, rec);
      }
    }
    
    return Array.from(deduped.values())
      .sort((a, b) => b.confidence - a.confidence);
  }
}

export default new ChartRecommender();
```

```javascript
    // 公式计算

      result = this.applyFormulaCalculation(result, transformConfig.formulaCalculation);
```

### 3.5 错误收集器（新增：errorCollector.js）

#### 核心功能

1. **自动捕获错误**
2. **本地存储**
3. **批量上报**

#### 实现代码

```javascript
// errorCollector.js
class ErrorCollector {
  constructor() {
    this.errors = [];
    this.storageKey = 'data_insight_errors';
    this.maxStorageSize = 1000; // 最多存储 1000 条
    
    // 从 localStorage 加载历史错误
    this.loadFromStorage();
    
    // 全局错误捕获
    this.setupGlobalErrorHandlers();
  }
  
  // 记录错误
  record(error, context = {}) {
    const errorRecord = {
      id: this.generateId(),
      timestamp: new Date().toISOString(),
      type: this.classifyError(error),
      error: {
        message: error.message || String(error),
        stack: error.stack || '',
        name: error.name || 'Error'
      },
      context: {
        userInput: context.userInput || '',
        intent: context.intent || '',
        module: context.module || '',
        action: context.action || ''
      },
      isSubmitted: false
    };
    
    this.errors.unshift(errorRecord); // 新错误在前
    
    // 限制存储数量
    if (this.errors.length > this.maxStorageSize) {
      this.errors = this.errors.slice(0, this.maxStorageSize);
    }
    
    // 保存到 localStorage
    this.saveToStorage();
    
    // 同步到全局错误日志
    if (window.globalErrorLog) {
      window.globalErrorLog.push(errorRecord);
    }
    
    console.error('[ErrorCollector] 记录错误:', errorRecord);
  }
  
  // 获取未提交的错误
  getUnsubmittedErrors() {
    return this.errors.filter(e => !e.isSubmitted);
  }
  
  // 标记为已提交
  markAsSubmitted(errorIds) {
    for (const error of this.errors) {
      if (errorIds.includes(error.id)) {
        error.isSubmitted = true;
        error.submittedAt = new Date().toISOString();
      }
    }
    this.saveToStorage();
  }
  
  // 清除错误
  clear(errorIds) {
    if (errorIds) {
      this.errors = this.errors.filter(e => !errorIds.includes(e.id));
    } else {
      this.errors = [];
    }
    this.saveToStorage();
  }
  
  // 导出错误报告
  exportReport() {
    return {
      exportTime: new Date().toISOString(),
      totalErrors: this.errors.length,
      errors: this.errors,
      statistics: this.generateStatistics()
    };
  }
  
  // 保存到 localStorage
  saveToStorage() {
    try {
      localStorage.setItem(this.storageKey, JSON.stringify(this.errors));
    } catch (e) {
      console.warn('保存到 localStorage 失败:', e);
    }
  }
  
  // 从 localStorage 加载
  loadFromStorage() {
    try {
      const stored = localStorage.getItem(this.storageKey);
      if (stored) {
        this.errors = JSON.parse(stored);
      }
    } catch (e) {
      console.warn('从 localStorage 加载失败:', e);
    }
  }
  
  // 生成统计信息
  generateStatistics() {
    const stats = {
      byType: {},
      byModule: {},
      byDate: {},
      recentTrends: []
    };
    
    for (const error of this.errors) {
      // 按类型统计
      stats.byType[error.type] = (stats.byType[error.type] || 0) + 1;
      
      // 按模块统计
      const module = error.context.module || 'unknown';
      stats.byModule[module] = (stats.byModule[module] || 0) + 1;
      
      // 按日期统计
      const date = error.timestamp.split('T')[0];
      stats.byDate[date] = (stats.byDate[date] || 0) + 1;
    }
    
    return stats;
  }
  
  // 生成唯一 ID
  generateId() {
    return `ERR_${Date.now()}_${Math.random().toString(36).substr(2, 9)}`;
  }
  
  // 错误分类
  classifyError(error) {
    if (error.message?.includes('column')) return 'COLUMN_ERROR';
    if (error.message?.includes('param')) return 'PARAM_ERROR';
    if (error.message?.includes('network')) return 'NETWORK_ERROR';
    if (error.message?.includes('timeout')) return 'TIMEOUT_ERROR';
    return 'UNKNOWN_ERROR';
  }
  
  // 设置全局错误捕获
  setupGlobalErrorHandlers() {
    window.addEventListener('error', (e) => {
      this.record(e.error || e, {
        module: 'global',
        action: 'uncaught_error'
      });
    });
    
    window.addEventListener('unhandledrejection', (e) => {
      this.record(e.reason || new Error('Unhandled rejection'), {
        module: 'global',
        action: 'unhandled_promise'
      });
    });
  }
}

export default new ErrorCollector();
```

### 3.6 用户反馈器（新增：feedbackManager.js）

#### 核心功能

1. **多类型反馈**：错误反馈、评分反馈、建议反馈
2. **快捷反馈入口**：一键反馈识别有误
3. **反馈激励**：感谢提示

#### 实现代码

```javascript
// feedbackManager.js
class FeedbackManager {
  constructor() {
    this.feedbackRecords = [];
    this.storageKey = 'data_insight_feedback';
    
    this.loadFromStorage();
  }
  
  // 提交反馈
  submit(feedback) {
    const record = {
      id: this.generateId(),
      timestamp: new Date().toISOString(),
      ...feedback,
      isSubmitted: false
    };
    
    this.feedbackRecords.unshift(record);
    this.saveToStorage();
    
    // 触发回调
    if (this.onFeedbackSubmitted) {
      this.onFeedbackSubmitted(record);
    }
    
    return record;
  }
  
  // 快捷反馈：识别有误
  quickFeedbackIncorrect(userInput, intentResult, reason) {
    return this.submit({
      type: 'error_report',
      category: 'intent_recognition',
      userInput,
      expectedIntent: reason,
      actualIntent: intentResult,
      severity: 'medium',
      description: `用户反馈识别错误：期望"${reason}"，实际"${intentResult.intent}"`
    });
  }
  
  // 评分反馈
  submitRating(score, category, comment = '') {
    return this.submit({
      type: 'rating',
      score, // 1-5
      category, // 'intent_accuracy', 'chart_quality', 'response_speed'
      comment
    });
  }
  
  // 建议反馈
  submitSuggestion(description, category = 'general') {
    return this.submit({
      type: 'suggestion',
      category,
      description
    });
  }
  
  // 获取反馈历史
  getHistory(limit = 50) {
    return this.feedbackRecords.slice(0, limit);
  }
  
  // 导出反馈数据
  exportData() {
    return {
      exportTime: new Date().toISOString(),
      totalFeedback: this.feedbackRecords.length,
      feedback: this.feedbackRecords,
      statistics: this.generateStatistics()
    };
  }
  
  // 生成统计
  generateStatistics() {
    const stats = {
      byType: {},
      byCategory: {},
      averageRating: 0,
      ratingDistribution: {}
    };
    
    let ratingSum = 0;
    let ratingCount = 0;
    
    for (const record of this.feedbackRecords) {
      // 按类型统计
      stats.byType[record.type] = (stats.byType[record.type] || 0) + 1;
      
      // 按类别统计
      const category = record.category || 'general';
      stats.byCategory[category] = (stats.byCategory[category] || 0) + 1;
      
      // 评分统计
      if (record.type === 'rating' && record.score) {
        ratingSum += record.score;
        ratingCount++;
        stats.ratingDistribution[record.score] = (stats.ratingDistribution[record.score] || 0) + 1;
      }
    }
    
    if (ratingCount > 0) {
      stats.averageRating = (ratingSum / ratingCount).toFixed(2);
    }
    
    return stats;
  }
  
  // 工具方法
  saveToStorage() {
    try {
      localStorage.setItem(this.storageKey, JSON.stringify(this.feedbackRecords));
    } catch (e) {
      console.warn('保存反馈失败:', e);
    }
  }
  
  loadFromStorage() {
    try {
      const stored = localStorage.getItem(this.storageKey);
      if (stored) {
        this.feedbackRecords = JSON.parse(stored);
      }
    } catch (e) {
      console.warn('加载反馈失败:', e);
    }
  }
  
  generateId() {
    return `FB_${Date.now()}_${Math.random().toString(36).substr(2, 9)}`;
  }
}

export default new FeedbackManager();
```

### 3.7 自动修复器（新增：autoFixer.js）

#### 核心功能

1. **规则引擎**：基于规则自动修复
2. **定时任务**：定时自动处理
3. **手动触发**：一键修复所有

#### 实现代码

```javascript
// autoFixer.js
class AutoFixer {
  constructor() {
    this.fixRules = [];
    this.isScheduled = false;
    this.fixHistory = [];
    
    this.registerDefaultRules();
  }
  
  // 注册默认修复规则
  registerDefaultRules() {
    // 规则 1：语义映射缺失
    this.registerRule({
      id: 'semantic_mapping_missing',
      name: '语义映射缺失自动补充',
      condition: (error) => {
        return error.type === 'COLUMN_ERROR' && 
               error.error.message?.includes('未能匹配到列名');
      },
      fixAction: (error) => {
        // 从错误上下文中提取新术语
        const userInput = error.context.userInput;
        // TODO: 调用语义映射更新逻辑
        return {
          success: true,
          action: 'added_semantic_mapping',
          details: `从"${userInput}"中提取新术语`
        };
      }
    });
    
    // 规则 2：意图识别置信度低
    this.registerRule({
      id: 'low_confidence_intent',
      name: '低置信度意图优化',
      condition: (error) => {
        return error.context.intent?.confidence < 0.5;
      },
      fixAction: (error) => {
        // TODO: 调整识别阈值或添加新规则
        return {
          success: true,
          action: 'adjusted_threshold',
          details: '降低置信度阈值'
        };
      }
    });
    
    // 规则 3：参数提取失败
    this.registerRule({
      id: 'param_extraction_failed',
      name: '参数提取失败修复',
      condition: (error) => {
        return error.type === 'PARAM_ERROR' &&
               error.error.message?.includes('参数提取失败');
      },
      fixAction: (error) => {
        // TODO: 添加新的参数提取模式
        return {
          success: true,
          action: 'added_extraction_pattern',
          details: '添加新的参数提取正则模式'
        };
      }
    });
  }
  
  // 注册修复规则
  registerRule(rule) {
    this.fixRules.push(rule);
  }
  
  // 执行自动修复
  async executeAutoFix(errorRecords) {
    const results = [];
    
    for (const error of errorRecords) {
      // 查找匹配的修复规则
      const matchedRules = this.fixRules.filter(rule => rule.condition(error));
      
      for (const rule of matchedRules) {
        try {
          const fixResult = await rule.fixAction(error);
          
          results.push({
            errorId: error.id,
            ruleId: rule.id,
            ruleName: rule.name,
            success: fixResult.success,
            action: fixResult.action,
            details: fixResult.details,
            timestamp: new Date().toISOString()
          });
          
          // 记录修复历史
          this.fixHistory.push({
            ...results[results.length - 1],
            error: error
          });
          
          if (fixResult.success) {
            break; // 修复成功后不再尝试其他规则
          }
        } catch (e) {
          console.error(`规则${rule.name}执行失败:`, e);
        }
      }
    }
    
    return results;
  }
  
  // 定时自动修复
  scheduleAutoFix(intervalMinutes = 60) {
    if (this.isScheduled) {
      clearInterval(this.scheduleTimer);
    }
    
    this.isScheduled = true;
    this.scheduleTimer = setInterval(async () => {
      console.log('[AutoFixer] 开始定时自动修复...');
      
      // 获取未修复的错误
      const errorCollector = window.errorCollector;
      if (errorCollector) {
        const errors = errorCollector.getUnsubmittedErrors();
        const results = await this.executeAutoFix(errors);
        
        console.log(`[AutoFixer] 完成修复：${results.length} 条`);
        
        // 标记已修复的错误
        const fixedErrorIds = results
          .filter(r => r.success)
          .map(r => r.errorId);
        
        if (fixedErrorIds.length > 0) {
          errorCollector.markAsSubmitted(fixedErrorIds);
        }
      }
    }, intervalMinutes * 60 * 1000);
    
    console.log(`[AutoFixer] 已启动定时任务，间隔：${intervalMinutes}分钟`);
  }
  
  // 手动触发修复
  async triggerFix(errorIds) {
    const errorCollector = window.errorCollector;
    if (!errorCollector) {
      throw new Error('错误收集器未初始化');
    }
    
    // 获取指定错误
    const errors = errorCollector.errors.filter(e => errorIds.includes(e.id));
    
    return await this.executeAutoFix(errors);
  }
  
  // 一键修复所有
  async fixAll() {
    const errorCollector = window.errorCollector;
    if (!errorCollector) {
      throw new Error('错误收集器未初始化');
    }
    
    const errors = errorCollector.getUnsubmittedErrors();
    console.log(`[AutoFixer] 开始修复所有 ${errors.length} 条错误...`);
    
    return await this.executeAutoFix(errors);
  }
  
  // 获取修复历史
  getFixHistory(limit = 100) {
    return this.fixHistory.slice(-limit);
  }
  
  // 获取修复统计
  getStatistics() {
    const stats = {
      totalFixes: this.fixHistory.length,
      successRate: 0,
      byRule: {},
      recentFixes: []
    };
    
    let successCount = 0;
    
    for (const fix of this.fixHistory) {
      if (fix.success) successCount++;
      
      // 按规则统计
      stats.byRule[fix.ruleId] = (stats.byRule[fix.ruleId] || 0) + 1;
    }
    
    if (this.fixHistory.length > 0) {
      stats.successRate = (successCount / this.fixHistory.length * 100).toFixed(2) + '%';
    }
    
    stats.recentFixes = this.fixHistory.slice(-10);
    
    return stats;
  }
}

export default new AutoFixer();
```
